374  void buildMessage(DoIPPayloadType payloadType, const ByteArray &payload) {
375  m_data.clear();
376  m_data.reserve(DOIP_HEADER_SIZE + payload.size());
377 
378  // Protocol version
379  m_data.emplace_back(PROTOCOL_VERSION);
380  m_data.emplace_back(PROTOCOL_VERSION_INV);
381 
382  // Payload type (big-endian uint16_t)
383  m_data.writeEnum(payloadType);
384 
385  // Payload length (big-endian uint32_t)
386  uint32_t payloadLength = static_cast<uint32_t>(payload.size());
387  m_data.writeU32(payloadLength);
388 
389  // Payload data
390  m_data.insert(m_data.end(), payload.begin(), payload.end());
391  }
