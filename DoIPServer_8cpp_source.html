165 ssize_t DoIPServer::receiveUdpMessage() {
166  // Set socket timeout to prevent blocking indefinitely
167  struct timeval timeout;
168  timeout.tv_sec = 1; // 1 second timeout
169  timeout.tv_usec = 0;
170  setsockopt(m_udp_sock, SOL_SOCKET, SO_RCVTIMEO, &timeout, sizeof(timeout));
171 
172  struct sockaddr_in clientAddr;
173  unsigned int length = sizeof(clientAddr);
174  const ssize_t readBytes = recvfrom(m_udp_sock, m_receiveBuf.data(), m_receiveBuf.size(), 0, reinterpret_cast<struct sockaddr *>(&clientAddr), &length);
175 
176  if (readBytes < 0) {
177  if (errno == EAGAIN) {
178  // Timeout - this is normal, just continue
179  return 0;
180  } else {
181  UDP_LOG_ERROR("Error receiving UDP message: {}", strerror(errno));
182  return -1;
183  }
184  }
185 
186  // Don't log if no data received (can happen with some socket configurations)
187  if (readBytes > 0) {
188  UDP_LOG_INFO("RX {} bytes from {}:{}", readBytes,
189  inet_ntoa(clientAddr.sin_addr), ntohs(clientAddr.sin_port));
190  } else {
191  // For debugging: log zero-byte messages at debug level
192  UDP_LOG_DEBUG("RX 0 bytes from {}:{}",
193  inet_ntoa(clientAddr.sin_addr), ntohs(clientAddr.sin_port));
194  return 0; // Return early for zero-byte messages
195  }
196 
197  // Store client address for response
198  m_clientAddress = clientAddr;
199 
200  ssize_t sentBytes = reactToReceivedUdpMessage(static_cast<size_t>(readBytes));
201 
202  return sentBytes;
203 }
